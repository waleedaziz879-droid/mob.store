-- Electra Mobile Store Schema
DROP TABLE IF EXISTS order_items CASCADE;
DROP TABLE IF EXISTS orders CASCADE;
DROP TABLE IF EXISTS products CASCADE;

CREATE TABLE products (
  id            TEXT PRIMARY KEY,
  name          TEXT NOT NULL,
  brand         TEXT NOT NULL,
  price         NUMERIC NOT NULL CHECK (price >= 0),
  rating        NUMERIC NOT NULL DEFAULT 5.0,
  reviews_count INTEGER NOT NULL DEFAULT 0,
  description   TEXT,
  colors        JSONB NOT NULL,
  storage       JSONB NOT NULL,
  specs         JSONB NOT NULL,
  featured      BOOLEAN DEFAULT FALSE,
  created_at    TIMESTAMPTZ DEFAULT now() NOT NULL
);

CREATE TABLE orders (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_name    TEXT NOT NULL,
  customer_email   TEXT NOT NULL,
  shipping_address TEXT NOT NULL,
  subtotal         NUMERIC NOT NULL,
  tax              NUMERIC NOT NULL,
  shipping         NUMERIC NOT NULL,
  total            NUMERIC NOT NULL,
  status           TEXT DEFAULT 'Pending'
                     CHECK (status IN ('Pending','Shipped','Delivered','Cancelled')),
  created_at       TIMESTAMPTZ DEFAULT now() NOT NULL
);

CREATE TABLE order_items (
  id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id     UUID REFERENCES orders(id) ON DELETE CASCADE NOT NULL,
  product_id   TEXT NOT NULL,
  product_name TEXT NOT NULL,
  color        TEXT NOT NULL,
  storage      TEXT NOT NULL,
  price        NUMERIC NOT NULL,
  quantity     INTEGER NOT NULL CHECK (quantity > 0)
);

ALTER TABLE products   ENABLE ROW LEVEL SECURITY;
ALTER TABLE orders     ENABLE ROW LEVEL SECURITY;
ALTER TABLE order_items ENABLE ROW LEVEL SECURITY;

CREATE POLICY "all_products"    ON products    FOR ALL TO public USING (true) WITH CHECK (true);
CREATE POLICY "all_orders"      ON orders      FOR ALL TO public USING (true) WITH CHECK (true);
CREATE POLICY "all_order_items" ON order_items FOR ALL TO public USING (true) WITH CHECK (true);
